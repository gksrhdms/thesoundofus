typedef struct {
  char    name[64];
  int     age;
  char   *questions[32];
  char   *answers[32];
  int     count;
} Interview;
void init(Interview *iv, char *name, int age) {
  strcpy(iv->name, name);
  iv->age   = age;
  iv->count = 0;
}
void ask(Interview *iv, char *q) {
  iv->questions[iv->count] = q;
  iv->answers[iv->count]   = NULL;
  iv->count++;
}
void answer(Interview *iv, int i, char *a) {
  iv->answers[i] = a;
}
void print(Interview *iv) {
  for (int i = 0; i < iv->count; i++) {
    printf("Q: %s\n", iv->questions[i]);
    printf("A: %s\n", iv->answers[i]);
  }
}
